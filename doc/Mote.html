def as_json(options = {})
	ports = self.ports
	sensors = {}
	transforms = {}

	ports.each do |port|
		if port.portNumber < 10
			prefix = 'A0'
		else
			prefix = 'A'
		end
		name = prefix + port.portNumber.to_s
		sensors[name] = port.sensor.name
		i = 0
		port.transforms.each do |transform|
			tranName = name + '_' + i.to_s
			i = i + 1
			transforms[tranName] = transform.name
		end
  end

	{ mote: {name: self.name, radio: self.radio.name, sensors: sensors, transforms: transforms}}
end
